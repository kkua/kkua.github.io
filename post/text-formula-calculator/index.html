import cn.hutool.core.lang.Pair;
import com.ruoyi.common.exception.ServiceException;
import com.ruoyi.common.utils.StringUtils;
import lombok.Getter;
import lombok.Setter;
import org.apache.commons.collections4.SetUtils;
import org.apache.commons.lang3.math.NumberUtils;

import java.math.BigDecimal;
import java.math.RoundingMode;
import java.util.*;
import java.util.regex.Pattern;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public class Formula {

    /**
     * 正规化公式
     */
    private String normalExpression = null;

    /**
     * 操作数栈，String或BigDecimal
     */
    private Stack<Pair<String, BigDecimal>> operandStack = new Stack<>();

    @Getter
    private Set<String> refVariables = new HashSet<>();

    /**
     * 左操作数
     */
    private Pair<String, BigDecimal> lhs;

    /**
     * 右操作数
     */
    private Pair<String, BigDecimal> rhs;

    /**
     * 公式等价后缀表达式（逆波兰表达式）
     */
    private List<String> postfixExpSeq;

    private String currentOp;

    private Iterator<String> cursor;

    private static int BRACKET_START = '(';

    private static int BRACKET_END = ')';

    private static int ADD_OP = '+';

    private static int SUB_OP = '-';

    private static int MUL_OP = '*';

    private static int DIV_OP = '/';

    private static List<Integer> OPERATORS = Arrays.asList(ADD_OP, SUB_OP, MUL_OP, DIV_OP);

    /**
     * 优先级
     */
    private static Map<Integer, Integer> OPERATOR_PRIO = Stream.of(new Object[][]{
            {ADD_OP, 100},
            {SUB_OP, 100},
            {MUL_OP, 200},
            {DIV_OP, 200},
    }).collect(Collectors.toMap(obj -> (Integer)obj[0], obj-> (Integer) obj[1]));

    private static final List<Integer> MATH_SYMBOLS = Stream.concat(OPERATORS.stream(), Stream.of(BRACKET_START, BRACKET_END))
            .collect(Collectors.toList());;

    @Getter
    private BigDecimal result = null;

    @Getter
    private boolean complete = false;

    @Getter
    @Setter
    private int nullResultCount;


    public static void checkVariableName(String variableName) {
        if (variableName.codePoints().anyMatch(i -> MATH_SYMBOLS.contains(i))) {
            throw new ServiceException("名称中不能包含字符：+-*/()");
        }
    }

    public List<String> parse() {
        return parse(normalExpression);
    }

    public List<String> parse(String expression) {
        PrimitiveIterator.OfInt it = expression.codePoints().iterator();
        List<String> tokenList = new ArrayList<>();
        List<Integer> operandChars = new ArrayList<>();

//   1）我们使用一个stack栈结构存储操作符，用一个List结构存储后缀表达式结果
//
//　　2）首先读取到数字，直接存入list中
//
//　　3）当读取到左括号"("时，直接压栈，当读取到运算符时，分两种情况讨论
//
//　　　　a.当运算符栈为空或者栈顶操作符的优先级小于当前运算符优先级时(如+和-的优先级低于 * 和 /)，直接入栈
//
//　　　　b.当运算符不为空时且栈顶操作符的优先级大于或等于当前运算符优先级时，循环执行出栈操作并加入list中，直到遇到优先级小于当前运算符的元素为止。循环执行完后再将当前运算符压栈。另外需要注意的是，只有遇到右括号时，左括号才出栈
//
//　　4) 当遇到右括号")"时，循环执行出栈操作并加入到list中，直到遇到左括号为止。并将左括号弹出，但不加入list中
//
//　　5) 表达式的值读取完后，将操作符栈中的所有元素弹出并加入到list中
        Stack<Integer> operatorStack = new Stack<>();
        while (it.hasNext()) {
            Integer currentChar = it.next();
            if (MATH_SYMBOLS.contains(currentChar)) {
                addOperandToken(tokenList, operandChars);
                if (currentChar == BRACKET_START) {
                    operatorStack.push(currentChar);
                } else if (OPERATORS.contains(currentChar)) {
                    while (true) {
                        if (operatorStack.isEmpty() || operatorStack.peek() == BRACKET_START
                                || OPERATOR_PRIO.get(operatorStack.peek()) < OPERATOR_PRIO.get(currentChar)) {
                            operatorStack.push(currentChar);
                            break;
                        } else {
                            tokenList.add(new String(Character.toChars(operatorStack.pop())));
                        }
                    }
                } else if (currentChar == BRACKET_END) {
                    int popped = operatorStack.pop().intValue();
                    while (popped != BRACKET_START) {
                        tokenList.add(new String(Character.toChars(popped)));
                        if (operatorStack.isEmpty()) {
                            break;
                        }
                        popped = operatorStack.pop().intValue();
                    }
                }
            } else {
                operandChars.add(currentChar);
            }
        }
        addOperandToken(tokenList, operandChars);
        while (!operatorStack.isEmpty()) {
            tokenList.add(new String(Character.toChars(operatorStack.pop())));
        }
        postfixExpSeq = tokenList;
        cursor = postfixExpSeq.iterator();
        return tokenList;
    }

    private void addOperandToken(List<String> tokenList, List<Integer> operandChars) {
        int operandLen = operandChars.size();
        if (operandLen > 0) {
            String str = new String(operandChars.stream().mapToInt(Integer::intValue).toArray(), 0, operandLen);
            if (StringUtils.isNotBlank(str)) {
                String token = str.trim();
                tokenList.add(token);
                if (!NumberUtils.isParsable(token)) {
                    // 变量
                    refVariables.add(token);
                }
            }
            operandChars.clear();
        }
    }

    private BigDecimal doCalculateAndCount(Map<String, BigDecimal> refValueMap, boolean isResume) {
        BigDecimal result = doCalculate(refValueMap, isResume);
        if (Objects.isNull(result)) {
            ++nullResultCount;
        }
        return result;

    }

    private BigDecimal doCalculate(Map<String, BigDecimal> refValueMap, boolean isResume) {
        if (Objects.isNull(lhs) || Objects.isNull(rhs)) {
            return null;
        }
        BigDecimal lhsValue = lhs.getValue();
        BigDecimal rhsValue = rhs.getValue();

//        if (isResume) {
            if (Objects.isNull(rhsValue) && refValueMap.containsKey(rhs.getKey())) {
                rhsValue = refValueMap.get(rhs.getKey());
                if (Objects.isNull(rhsValue)) {
                    result = null;
                    turn2Complete();
                    return null;
                }
            }
            if (Objects.isNull(lhsValue) && refValueMap.containsKey(lhs.getKey())) {
                lhsValue = refValueMap.get(lhs.getKey());
                if (Objects.isNull(lhsValue)) {
                    result = null;
                    turn2Complete();
                    return null;
                }
            }
//        }
        if (Objects.isNull(lhsValue) || Objects.isNull(rhsValue)) {
            return null;
        }
        BigDecimal result = BigDecimal.ZERO;
        switch (currentOp) {
            case "+":
                result = lhsValue.add(rhsValue);
                break;
            case "-":
                result= lhsValue.subtract(rhsValue);
                break;
            case "*":
                result = lhsValue.multiply(rhsValue);
                break;
            case "/":
                result = lhsValue.divide(rhsValue, 10, RoundingMode.HALF_UP);
                break;
        }
        this.result = result;
        resetStatus();
        return result;
    }

    public boolean calculate(Map<String, BigDecimal> refValueMap) {
        if (StringUtils.isNotBlank(currentOp)) {
            result = doCalculateAndCount(refValueMap, true);
            if (complete) {
                return true;
            }
            if (Objects.isNull(result)) {
                return false;
            }
            operandStack.push(Pair.of(null, result));
        }
        while (!complete && cursor.hasNext()) {
            String token = cursor.next();
            if ("+-*/".contains(token)) {
                // 运算符
                currentOp = token;
                rhs = operandStack.pop();
                lhs = operandStack.pop();

                BigDecimal lhsValue = lhs.getValue();
                BigDecimal rhsValue = rhs.getValue();
                String rhsLabel = rhs.getKey();
                String lhsLabel = lhs.getKey();
                if (Objects.isNull(rhsValue) && NumberUtils.isParsable(rhsLabel)) {
                    rhsValue = new BigDecimal(rhsLabel);
                    rhs = Pair.of(rhsLabel, rhsValue);
                }
                if (Objects.isNull(lhsValue) && NumberUtils.isParsable(lhsLabel)) {
                    lhsValue = new BigDecimal(lhsLabel);
                    lhs = Pair.of(lhsLabel, lhsValue);
                }
                result = doCalculateAndCount(refValueMap, false);
                if (complete) {
//                    终止计算
                    break;
                }
                if (Objects.isNull(result)) {
                    return false;
                }
                resetStatus();
                operandStack.push(Pair.of(null, result));
            } else {
                // 入栈
                operandStack.push(Pair.of(token, null));
            }
        }
        // 特例：单操作数的情况
        if (postfixExpSeq.size() == 1) {
            Pair<String, BigDecimal> operand = operandStack.pop();
            String str = operand.getKey();
            if (NumberUtils.isParsable(str)) {
                result = new BigDecimal(str);
            } else if (refValueMap.containsKey(str)) {
                result = refValueMap.get(str);
            } else {
                operandStack.push(operand);
                return false;
            }
        }
        turn2Complete();
        return true;
    }

    private void turn2Complete() {
        complete = true;
        resetStatus();
        operandStack.clear();
    }
    private void resetStatus() {
        currentOp = null;
        lhs = null;
        rhs = null;
        nullResultCount = 0;
    }

    public void reset4Reuse() {
        complete = false;
        result = null;
        operandStack.clear();
        cursor = postfixExpSeq.iterator();
        resetStatus();
    }

    /**
     * 使用正则来校验数学公式
     * <br>
     * 对公式进行校验，除了一些常规性的检查之外，还需要对公式的格式进行优化，为后续的公式计算扫清障碍。
     * <br>
     * 优化内容包括：
     * <br>
     * 对负数和正数进行补零，例如 -a 优化为 0-a。
     * 补充两个括号之间的乘法运算符，例如 (a+b)(b+c) 优化为 (a+b)*(b+c)。
     */
    public String validate(List<String> tokenList, Set<String> variables) {
        String expression = String.join("", tokenList);
        variables = SetUtils.emptyIfNull(variables);
        // 连续运算符处理
        if (expression.split("[\\+\\-\\*\\/]{2,}").length > 1) {
            throw new ServiceException("公式不合法，包含连续运算符");
        }
        if (StringUtils.contains(expression, "()")) {
            throw new ServiceException("公式不合法，包含空括号");
        }
        expression = expression.replaceAll("\\)\\(", "\\)*\\(");
        expression = expression.replaceAll("\\(\\-", "\\(0-");
        expression = expression.replaceAll("\\(\\+", "\\(0+");
        // 校验变量
        for (String variable : refVariables) {
            if (variable.length() == 1 && MATH_SYMBOLS.contains(variable.codePointAt(0))) {
                continue;
            }
            if (StringUtils.isBlank(variable) || NumberUtils.isParsable(variable)) {
                continue;
            }
            if (!variables.contains(variable)) {
                throw new ServiceException("公式不合法，字段【" + variable + "】不存在");
            }
        }
        // 校验括号
        Integer preChar = null;
        Stack<Integer> stack = new Stack<>();
        String resultExpression = expression;
        for (int i = 0; i < expression.length(); i++) {
            int currChar = expression.codePointAt(i);
            if (i == 0) {
                if (Pattern.matches("\\*|\\/", String.valueOf(currChar))) {
                    throw new ServiceException("公式不合法，以错误运算符开头");
                }
                if (currChar == ADD_OP) {
                    resultExpression = expression.substring(1);
                }
                if (currChar == SUB_OP) {
                    resultExpression = "0" + expression;
                }
            }
            if (BRACKET_START == currChar) {
                stack.push(currChar);
            } else if (BRACKET_END == currChar) {
                if (stack.size() > 0) {
                    stack.pop();
                } else {
                    throw new ServiceException("公式不合法，括号不配对");
                }
            }
            if (preChar != null && preChar == BRACKET_START && OPERATORS.contains(currChar)) {
                throw new ServiceException("公式不合法，左括号后是运算符");
            }
            if (preChar != null && preChar == BRACKET_END && !OPERATORS.contains(currChar)) {
                throw new ServiceException("公式不合法，右括号后面不是运算符");
            }
            if (i == expression.length() - 1) {
                if (OPERATORS.contains(currChar))
                    throw new ServiceException("公式不合法，以运算符结尾");
            }
            preChar = currChar;
        }
        if (stack.size() > 0) {
            throw new ServiceException("公式不合法，括号不配对");
        }
        return resultExpression;
    }

    public String normalize(String fText, Set<String> variables) {
        List<String> list = normalize(fText);
        String expression = validate(list, variables);
        this.normalExpression = expression;
        return expression;
    }

    /**
     * 规格化表达式
     * @param fText
     * @return
     */
    public List<String> normalize(String fText) {
        PrimitiveIterator.OfInt it = fText.codePoints().iterator();
        List<String> tokenList = new ArrayList<>();
        List<Integer> operandChars = new ArrayList<>();
        while (it.hasNext()) {
            Integer currentChar = it.next();
            int tokenNum = tokenList.size();
            String prevToken = "";
            if (tokenNum > 0) {
                prevToken = tokenList.get(tokenNum - 1);
            }
            if (MATH_SYMBOLS.contains(currentChar)) {
                boolean noOperand = operandChars.isEmpty() || operandChars.stream().allMatch(ch -> Character.isWhitespace(ch));
                if (noOperand && prevToken.length() == 1 && prevToken.codePointAt(0) == BRACKET_START) {
//                if (StringUtils.equals(prevToken, "(")) {
                    if (Arrays.asList(ADD_OP, SUB_OP).contains(currentChar)) {
                        tokenList.add("0");
                    } else if (currentChar == BRACKET_START) {
                        tokenList.add("(");
                    } else if (currentChar == BRACKET_END) {
                        throw new ServiceException("公式不合法，包含空括号");
                    } else {
                        throw new ServiceException("公式不合法，左括号后是运算符");
                    }
                } else {
                    addOperandToken(tokenList, operandChars);
                    tokenList.add(new String(Character.toChars(currentChar)));
                }
            } else {
                operandChars.add(currentChar);
            }
        }
        addOperandToken(tokenList, operandChars);
        return tokenList;
    }

}

多公式关联计算

有向图环路检测

一个公式计算结果可以被其他公式作为变量引用，所以需要检测公式间是否存在循环应用。这是典型的有向图环路检测问题。下面介绍2种环路检测算法

  1. 深度优先遍历（DFS） 若在深度优先搜索的过程中遇到回边(即指向已经访问过的顶点的边)，则必定存在环。具体来说：对每个节点进行一轮深度遍历，在遍历的过程中，一旦发现当前访问节点在本轮遍历中已经访问过时，就说明图存在环路

  2. 拓扑排序 图的拓扑排序是针对有向无环图（DAG，Directed Acyclic Graph）的一种排序方式，它将图中的所有顶点排成一个线性序列，使得对于任何一条有向边 (u, v)，顶点 u 都在顶点 v 的前面。这种排序不是唯一的。 拓扑排序的算法步骤： 计算图中所有顶点的入度：入度是指有多少条边指向该顶点。

将所有入度为0的顶点加入到一个队列（或栈）中：这些顶点在拓扑排序的序列中可以被放置在最前面，因为它们没有依赖于其他顶点。

循环以下步骤直到队列为空：

  1. 从队列中移除一个顶点，将其加入到拓扑排序的结果序列中。

  2. 遍历该顶点的所有邻接点，将每个邻接点的入度减1。如果某个邻接点的入度变为0，则将其加入队列中。

  3. 如果所有顶点都被加入到拓扑排序序列中，则算法成功结束。

  4. 如果队列非空但在某个时刻无法继续执行（即仍有顶点在队列外），则说明原图不是有向无环图，不能进行拓扑排序。

  5. 不断在有向图中找出入度为0的节点，删除这些节点和其相连的边，直至全部节点都被删除或者图不为空且找不到一个入度为0的节点。遍历停止时图不为空则说明存在环路

import cn.hutool.core.lang.Pair;
import com.ruoyi.common.exception.ServiceException;
import com.ruoyi.common.utils.StringUtils;
import org.apache.commons.collections4.CollectionUtils;

import java.math.BigDecimal;
import java.time.Duration;
import java.util.*;

/**
 * 公式集
 */
public class FormulaSet {

    private Map<String, Formula> formulaMap = new HashMap<>();

    public void addFormula(String variableName, String formulaText, Set<String> variables) {
        Formula f = new Formula();
        String expression = f.normalize(formulaText, variables);
        f.parse(expression);
        formulaMap.put(variableName, f);
    }

    public void init() {
        checkCycleRefer();
    }

    public void calculate(Map<String, BigDecimal> refValueMap) {
        checkCycleRefer();
        Queue<Map.Entry<String, Formula>> workQueue = new LinkedList<>();
        for (Map.Entry<String, Formula> entry : formulaMap.entrySet()) {
            workQueue.offer(entry);
        }
        Map.Entry<String, Formula> entry = workQueue.poll();

        while (Objects.nonNull(entry)) {
            Formula f = entry.getValue();
            if (f.calculate(refValueMap)) {
                // 计算完成
                refValueMap.put(entry.getKey(), f.getResult());
            } else {
                if (f.getNullResultCount() > formulaMap.size()) {
                    // 有依赖变量为null无法计算
                    continue;
                }
                workQueue.offer(entry);
            }
            // 未计算完成
            entry = workQueue.poll();
        }
    }

    public void checkCycleRefer() {
        // 入度表
        Map<String, Integer> indMap = new HashMap<>();
        for (Map.Entry<String, Formula> entry : formulaMap.entrySet()) {
            Formula f = entry.getValue();
            Set<String> refVariables = f.getRefVariables();
            indMap.putIfAbsent(entry.getKey(), 0);
            for (String node : refVariables) {
                if (formulaMap.containsKey(node)) {
                    int d = indMap.getOrDefault(node, 0);
                    indMap.put(node, d + 1);
                }
            }
        }
        while (!indMap.isEmpty()) {
            boolean hasZeroNode = false;
            for (String findZeroNode : indMap.keySet()) {
                int ind = indMap.get(findZeroNode);
                if (ind != 0) {
                    continue;
                }
                hasZeroNode = true;
                Formula f = formulaMap.get(findZeroNode);
                Set<String> refVariables = f.getRefVariables();
                // 从有向图中删除入度为0节点和与之相连的边（即后继节点入度减1）
                for (String node : refVariables) {
                    indMap.computeIfPresent(node, (k, v) -> v - 1);
                }
                indMap.remove(findZeroNode);
                break;
            }
            if (!hasZeroNode) {
                // 存在循环引用
                throw new ServiceException("公式间存在循环引用，无法计算");
            }
        }
    }

    /**
     * 构造需重新计算的公式集，依赖变化变量的其他公式
     * @param changedVariable  变化了的公式对应的变量名
     * @param affectedVariables  受到影响的变量，即需要重新计算的变量
     * @return
     */
    public FormulaSet buildRecalcWhenFormulaChanged(String changedVariable, Collection<String> affectedVariables)  {
        FormulaSet newSet = new FormulaSet();
        for (Map.Entry<String, Formula> entry : formulaMap.entrySet()) {
            Formula formula = entry.getValue();
            Set<String> refSet = formula.getRefVariables();
            String varName = entry.getKey();
            if (refSet.contains(changedVariable) || StringUtils.equals(varName, changedVariable)) {
                newSet.formulaMap.put(varName, formula);
                affectedVariables.add(varName);
            }
        }
        return newSet;
    }

    public Pair checkCanDeleteVariable(String variableName) {
        for (Map.Entry<String, Formula> entry : formulaMap.entrySet()) {
            String formulaName = entry.getKey();
            Formula formula = entry.getValue();
            Set<String> refVariables = formula.getRefVariables();
            if (CollectionUtils.isEmpty(refVariables)) {
                continue;
            }
            if (refVariables.contains(variableName)) {
                return Pair.of(false, variableName);
            }
        }
        return Pair.of(true, "允许删除");
    }

    public void resetStatus() {
        for (Formula formula : formulaMap.values()) {
            formula.reset4Reuse();
        }
    }
